     /*  [+380 mylist.cpp] */
 381 
 382 {
 383 
 384     T *i=l, *j=r;
 385     T y;
 386     T &x= *(l+((r-l)>>1));
 387     if (l>=r) return;
 388     if (r-l<SORT_THRESHOLD)
 389     {
 390         for(i=l;i<r;++i)
 391           for (j=i+1;j<r;++j)
 392             if (__comp<T,less>(*i, *j))
 393             {
 394                 y=*i;
 395                 *i=*j;
 396                 *j=y;
 397             }
 398         return;
 399     }
 400     do
 401     {
 402         while (__comp<T,less>(*i,x)) ++i;
 403         while (__comp<T,less>(x,*j)) --j;
 404         if (i<=j)
 405         {
 406             y=*i;
 407             *i=*j;
 408             *j=y;
 409             ++i;
 410             --j;
 411         }
 412     } while (i<=j);
 413     if (l<j) sort_impl<less>(l, j);
 414     if (i<r) sort_impl<less>(i, r);
