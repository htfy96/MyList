 442 template<bool less>
 443 void MyList<T>::sort_impl(T* l, T* r) __MYLIST_NOEXCEPT_IF_2(__MYLIST_COPY, __MYLIST_CONSTRUCT)
     /*  [+443 mylist.cpp] */
 444 {
 445 
 446     if (l>=r) return;
 447     switch(r-l)
 448     {
 449         case 1: return;
 450         case 2:
 451                 if(__comp<T,less>(*r,*l)) __swap(*l,*r);
 452                 return;
 453         case 3:
 454                 if (__comp<T,less>(*(l+1), *l)) __swap(*l, *(l+1));
 455                 if (__comp<T,less>(*r, *(l+1))) __swap(*(l+1), *r);
 456                 return;
 457         case 4:
 458                 if (__comp<T,less>(*(l+1), *l)) __swap(*l, *(l+1));
 459                 if (__comp<T,less>(*(l+2), *(l+1))) __swap(*(l+2), *(l+1));
 460                 if (__comp<T,less>(*(l+2), *r)) __swap (*(l+2),*r);
 461                 return;                
 462         default:
 463                 T *i=l, *j=r;
 464                 T &x= *(l+((r-l)>>1));
 465                 if (r-l<SORT_THRESHOLD)
 466                 {
 467                     for(i=l;i<r;++i)
 468                       for (j=i+1;j<r;++j)
 469                         if (__comp<T,less>(*j, *i))
 470                           __swap(*i, *j);
 471                     return;
 472                 }
 473                 do
 474                 {
 475                     while (__comp<T,less>(*i,x)) ++i;
 476                     while (__comp<T,less>(x,*j)) --j;
 477                     if (i<=j)
 478                     {
 479                         __swap(*i, *j);
 480                         ++i;
 481                         --j;
 482                     }
 483                 } while (i<=j);
 484                 if (l<j) sort_impl<less>(l, j);
 485                 if (i<r) sort_impl<less>(i, r);
